exponential_func <- function(reps) {
  result_list <- c(rep(NA,reps))
  for (i in 1:reps){
    if (i %% 2 == 0) {
      result_list[i] <- square_function(i)
    } else {
      result_list[i] <- cube_function(i)
    }
  }
  q <- max(result_list)
}
start.time <- Sys.time()
first <- exponential_func(1e4)
first
## [1] 9.997e+11
end.time <- Sys.time()
time_taken = end.time - start.time
